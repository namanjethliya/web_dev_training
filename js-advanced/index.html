Error Handling

Errors are inevitable in any program. JavaScript gives you tools to catch them gracefully, recover from them, and even create your own custom error types — so your app never just silently dies.

try / catch / finally

Wrap code that might fail in a try block. If it throws, the catch block runs. The finally block always runs — whether or not there was an error.

app.js
try {
  // Code that might throw
  const result = JSON.parse("{ invalid json }");
  console.log(result);
} catch (error) {
  // Runs only when an error is thrown
  console.log("Caught:", error.message);
} finally {
  // Always runs — use for cleanup
  console.log("Done — this always prints");
}
// Output:
// Caught: Unexpected token 'i', "{ invalid json }" is not valid JSON
// Done — this always prints

throw — Raise Your Own Errors

You can throw any value, but best practice is to throw an instance of Error (or a subclass). This preserves the stack trace and message.

app.js
function divide(a, b) {
  if (b === 0) {
    throw new Error("Division by zero is not allowed");
  }
  return a / b;
}

try {
  console.log(divide(10, 2));   // 5
  console.log(divide(10, 0));   // throws!
} catch (e) {
  console.log("Error:", e.message);
}
// Output:
// 5
// Error: Division by zero is not allowed

Built-in Error Types

Error TypeWhen it occursExample
Error Generic base — use when no specific type fits throw new Error("oops")
TypeError Operation on wrong type null.property
RangeError Value outside allowable range new Array(-1)
ReferenceError Accessing an undeclared variable console.log(x) (x not declared)
SyntaxError Invalid JavaScript syntax JSON.parse("bad")
URIError Malformed URI passed to decodeURIComponent decodeURIComponent("%")

Custom Error Classes

Extend Error to create domain-specific error types. Then you can catch different errors differently using instanceof.

custom-errors.js
class ValidationError extends Error {
  constructor(message, field) {
    super(message);
    this.name    = "ValidationError";
    this.field   = field;   // custom property
  }
}

class NetworkError extends Error {
  constructor(message, statusCode) {
    super(message);
    this.name       = "NetworkError";
    this.statusCode = statusCode;
  }
}

function validateAge(age) {
  if (typeof age !== "number") {
    throw new ValidationError("Age must be a number", "age");
  }
  if (age < 0 || age > 150) {
    throw new ValidationError("Age out of range", "age");
  }
}

try {
  validateAge("twenty");
} catch (e) {
  if (e instanceof ValidationError) {
    console.log(`Validation failed on field "${e.field}": ${e.message}`);
  } else if (e instanceof NetworkError) {
    console.log(`Network error ${e.statusCode}: ${e.message}`);
  } else {
    throw e;   // re-throw unknown errors
  }
}
// Output: Validation failed on field "age": Age must be a number
💡

Re-throwing: If you catch an error but it's not the type you expected, re-throw it with throw e. Swallowing every error silently is one of the most common mistakes in error handling — it hides real bugs.

Try It — Error Handling

Try it
Golden rule: Only catch errors you can meaningfully handle. If you don't know how to recover from an error, re-throw it. Silent catch blocks that do nothing are worse than no error handling at all.

Regular Expressions

Regular expressions (regex) are patterns that match text. They let you validate inputs, search strings, extract data, and replace text — all with compact, powerful syntax.

Creating a Regex

Two ways to create a regex
// Literal syntax — preferred for known patterns
const pattern = /hello/;

// Constructor — useful for dynamic patterns built from variables
const word = "hello";
const dynamic = new RegExp(word, "i");  // case-insensitive

Character Classes & Quantifiers

Character Classes

\d — digit (0–9)
\w — word char (a-z, A-Z, 0-9, _)
\s — whitespace (space, tab, newline)
[abc] — any of a, b, or c
[^abc] — anything except a, b, c
[a-z] — any lowercase letter
. — any character except newline

Quantifiers

* — 0 or more
+ — 1 or more
? — 0 or 1 (optional)
{3} — exactly 3
{2,5} — 2 to 5
{2,} — 2 or more
^ / $ — start / end of string

Flags

g — global (find all matches)
i — case-insensitive
m — multiline (^ and $ match line start/end)
s — dotAll (. matches newline too)

Key Methods

regex-methods.js
const email = "user@example.com";
const phone = "9876543210";

// .test() — returns true/false — use for validation
const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
console.log(emailRegex.test(email));       // true
console.log(emailRegex.test("notvalid"));  // false

// .match() — returns array of matches or null
const str = "Call 9876543210 or 9123456789 for details";
const phones = str.match(/\d{10}/g);       // g flag = all matches
console.log(phones);  // ["9876543210", "9123456789"]

// .exec() — like match but iterates, includes capture groups
const dateStr = "Today is 2024-06-15";
const dateRegex = /(\d{4})-(\d{2})-(\d{2})/;
const match = dateRegex.exec(dateStr);
if (match) {
  console.log("Full:", match[0]);   // "2024-06-15"
  console.log("Year:", match[1]);   // "2024"
  console.log("Month:", match[2]);  // "06"
  console.log("Day:", match[3]);    // "15"
}

// .replace() — substitute matches
const messy = "Hello   World  with  extra   spaces";
const clean = messy.replace(/\s+/g, " ");
console.log(clean);  // "Hello World with extra spaces"

// Replace with a function
const names = "john doe, jane smith";
const titled = names.replace(/\b\w/g, ch => ch.toUpperCase());
console.log(titled);  // "John Doe, Jane Smith"

// .split() with regex
const csv = "apple, banana,cherry ,  date";
const fruits = csv.split(/\s*,\s*/);
console.log(fruits);  // ["apple", "banana", "cherry", "date"]

Common Validation Patterns

PatternRegexWhat it validates
Email /^[^\s@]+@[^\s@]+\.[^\s@]+$/ Has @ and domain — basic check
Indian mobile /^[6-9]\d{9}$/ Starts with 6-9, 10 digits total
PIN code /^\d{6}$/ Exactly 6 digits
Alphanumeric /^[a-zA-Z0-9]+$/ Letters and numbers only
Hex colour /^#([0-9a-fA-F]{3}|[0-9a-fA-F]{6})$/ #fff or #ffffff format
URL (simple) /^https?:\/\/.+\..+/ Starts with http:// or https://

Local Storage & Session Storage

Browsers give you two simple key-value stores to persist data on the user's device — no server, no database needed. Perfect for preferences, form state, and small datasets.

localStorage vs sessionStorage

localStorage

Data persists until explicitly cleared. Survives tab closes, browser restarts, and computer reboots. Shared across all tabs of the same origin.

sessionStorage

Data lasts only for the browser tab session. Cleared when the tab is closed. Each tab gets its own isolated storage — not shared.

The Core API

storage.js
// setItem — save a string value
localStorage.setItem("theme", "dark");
localStorage.setItem("username", "Naman");

// getItem — retrieve (returns null if key doesn't exist)
const theme = localStorage.getItem("theme");    // "dark"
const missing = localStorage.getItem("xyz");    // null

// removeItem — delete one key
localStorage.removeItem("username");

// clear — wipe everything for this origin
// localStorage.clear();  // ← be careful!

// length and key() — iterate all stored keys
console.log(localStorage.length);
for (let i = 0; i < localStorage.length; i++) {
  const key = localStorage.key(i);
  console.log(key, "=", localStorage.getItem(key));
}

// sessionStorage works identically — just different scope
sessionStorage.setItem("tempData", "only for this tab");

Storing Objects — JSON.stringify / JSON.parse

localStorage only stores strings. To store objects or arrays, serialize them with JSON.stringify() and deserialize with JSON.parse().

storage-objects.js
// Saving an object
const user = { name: "Naman", age: 25, role: "student" };
localStorage.setItem("user", JSON.stringify(user));

// Retrieving — JSON.parse converts string back to object
const stored = localStorage.getItem("user");
if (stored) {
  const parsed = JSON.parse(stored);
  console.log(parsed.name);  // "Naman"
  console.log(parsed.age);   // 25
}

// Saving an array
const scores = [78, 85, 90, 62];
localStorage.setItem("scores", JSON.stringify(scores));
const loadedScores = JSON.parse(localStorage.getItem("scores"));
console.log(loadedScores);   // [78, 85, 90, 62]

// Helper functions — wrap with error handling
function saveToStorage(key, value) {
  try {
    localStorage.setItem(key, JSON.stringify(value));
  } catch (e) {
    console.error("Storage failed:", e.message);
  }
}

function loadFromStorage(key, defaultValue = null) {
  try {
    const item = localStorage.getItem(key);
    return item !== null ? JSON.parse(item) : defaultValue;
  } catch (e) {
    console.error("Parse failed:", e.message);
    return defaultValue;
  }
}

saveToStorage("prefs", { fontSize: 16, lang: "en" });
const prefs = loadFromStorage("prefs", { fontSize: 14, lang: "en" });
console.log(prefs.fontSize);  // 16

Storage Events

When localStorage changes in one tab, other tabs of the same origin receive a storage event. Useful for syncing state across tabs.

Listening for storage changes in other tabs
// This fires in OTHER tabs — not the tab that made the change
window.addEventListener("storage", (event) => {
  console.log("Key changed:", event.key);
  console.log("Old value:", event.oldValue);
  console.log("New value:", event.newValue);
  console.log("From origin:", event.url);
});

Try It — Local Storage

localStorage is fully accessible here. Values you set will persist in your browser's storage. Open DevTools → Application → Local Storage to see them.

Try it
⚠️

Limits & gotchas: Each origin gets ~5 MB of localStorage. In private/incognito mode, storage may be blocked or cleared on exit. Always wrap localStorage calls in try/catch — users can disable storage, or you might exceed the quota.

When to use which: Use localStorage for user preferences, cached API responses, and anything that should survive a browser restart. Use sessionStorage for temporary state like multi-step form data within a single tab session.

Debugging Like a Pro

Debugging is a core developer skill — not just fixing bugs, but knowing how to find them fast. The browser DevTools and the console object give you everything you need.

Beyond console.log — The Full console API

console-methods.js
// console.table — display arrays/objects as a table
const users = [
  { name: "Naman", age: 25, role: "admin" },
  { name: "Priya", age: 22, role: "user"  },
  { name: "Raj",   age: 30, role: "mod"   }
];
console.table(users);  // renders a formatted table in DevTools

// console.group / console.groupEnd — collapsible groups
console.group("User Validation");
  console.log("Checking name...");
  console.log("Checking email...");
  console.warn("Email not verified");
console.groupEnd();

// console.time / console.timeEnd — measure execution time
console.time("loop");
let sum = 0;
for (let i = 0; i < 1_000_000; i++) sum += i;
console.timeEnd("loop");  // "loop: 3.41ms"

// console.count — count how many times a label is reached
function processItem(type) {
  console.count(type);
}
processItem("A");  // A: 1
processItem("B");  // B: 1
processItem("A");  // A: 2

// console.assert — only logs when condition is false
console.assert(1 === 1, "Math is broken");  // nothing printed
console.assert(1 === 2, "1 is not 2!");     // prints assertion error

// console.dir — inspect DOM node properties
// console.dir(document.body);  // shows all properties/methods

// console.trace — show the call stack
function a() { b(); }
function b() { console.trace("called from b"); }
a();  // prints: called from b → b → a

The debugger Statement

Place debugger anywhere in your code. When DevTools is open, execution pauses there — exactly like a manual breakpoint. No need to find the line in the Sources panel.

app.js
function calculateTotal(cart) {
  debugger;  // execution pauses here when DevTools is open
  let total = 0;
  for (const item of cart) {
    total += item.price * item.qty;
  }
  return total;
}
// Open DevTools (F12) first, then call this function
// You can inspect cart, item, total in the Scope panel

DevTools — Key Panels for Debugging

Console Panel

Run JS interactively, see logs, errors, and warnings. Filter by level. Click file:line links to jump to source.

Sources Panel

Browse your files. Click a line number to set a breakpoint. Execution stops there — inspect variables in the Scope sidebar.

Call Stack Panel

Shows the chain of function calls that led to the current pause. Click any frame to jump to that function's context.

Watch Expressions

Add expressions (e.g., user.age > 18, cart.length) to the Watch panel. They re-evaluate at every breakpoint pause.

🛠️

Breakpoint workflow: (1) Open DevTools → Sources. (2) Click a line number to add a breakpoint. (3) Trigger the code (click a button, reload). (4) Execution pauses. (5) Use Step Over (F10) to go line by line, Step Into (F11) to enter a function call. (6) Check Scope for local variables, Watch for your expressions.

Debugging mindset: Don't guess — verify. When something isn't right, add console.log() before and after the suspect code to narrow down where the value changes. Binary search your way to the bug.

Web APIs — Browser Superpowers

The browser ships with a rich set of built-in APIs. You don't install them — they're already there. GPS, clipboard access, notifications — all through JavaScript.

Geolocation API

Get the user's physical location (requires permission + HTTPS in production).

geolocation.js — requires HTTPS and user permission
if ("geolocation" in navigator) {
  // Always check for support first!
  navigator.geolocation.getCurrentPosition(
    // Success callback
    (position) => {
      const lat = position.coords.latitude;
      const lon = position.coords.longitude;
      const acc = position.coords.accuracy;  // meters
      console.log(`Location: ${lat}, ${lon} (±${acc}m)`);
    },
    // Error callback
    (error) => {
      switch (error.code) {
        case error.PERMISSION_DENIED:
          console.log("User denied location access");
          break;
        case error.POSITION_UNAVAILABLE:
          console.log("Location unavailable");
          break;
        case error.TIMEOUT:
          console.log("Request timed out");
          break;
      }
    },
    // Options
    { timeout: 10000, enableHighAccuracy: true }
  );
} else {
  console.log("Geolocation not supported in this browser");
}

// Watch for continuous updates (e.g. for navigation)
const watchId = navigator.geolocation.watchPosition((pos) => {
  console.log("Updated position:", pos.coords.latitude);
});
// Stop watching
// navigator.geolocation.clearWatch(watchId);

Clipboard API

Read from and write to the system clipboard. Requires HTTPS and user gesture (a click). The modern async API is much cleaner than the old document.execCommand.

clipboard.js — requires HTTPS and user gesture
// Write to clipboard (e.g., "Copy to clipboard" button)
async function copyToClipboard(text) {
  try {
    await navigator.clipboard.writeText(text);
    console.log("Copied:", text);
  } catch (e) {
    console.error("Copy failed:", e.message);
  }
}

// Read from clipboard (requires explicit user permission)
async function pasteFromClipboard() {
  try {
    const text = await navigator.clipboard.readText();
    console.log("Pasted:", text);
    return text;
  } catch (e) {
    console.error("Paste failed:", e.message);
  }
}

// In a button handler:
document.getElementById("copyBtn").addEventListener("click", () => {
  copyToClipboard("https://example.com/share/abc123");
});

Notifications API

Show desktop notifications to the user. Requires permission — and HTTPS in production. Never request permission without a clear user action.

notifications.js — requires HTTPS and permission
async function requestAndNotify(title, body) {
  // Check support
  if (!("Notification" in window)) {
    console.log("Notifications not supported");
    return;
  }

  // Request permission (only ask after a user action — e.g., a click)
  const permission = await Notification.requestPermission();

  if (permission === "granted") {
    const notification = new Notification(title, {
      body:  body,
      icon:  "/icon.png",   // optional
      badge: "/badge.png",  // optional
      tag:   "update-1"     // replaces previous notification with same tag
    });

    // Handle click on the notification
    notification.onclick = () => {
      window.focus();
      notification.close();
    };
  } else {
    console.log("Permission:", permission);  // "denied" or "default"
  }
}

// Only call inside a button click handler
// requestAndNotify("New Message", "Naman sent you a file");

Other Useful Browser APIs

Fetch API

Load data from URLs without page refresh. fetch(url).then(r => r.json()) — covered deeply in the Async chapter.

Web Storage API

localStorage and sessionStorage — covered in Chapter 03.

Intersection Observer

Detect when an element scrolls into view — perfect for lazy loading images and scroll animations.

Web Workers

Run heavy computation in a background thread without blocking the UI. Use new Worker('worker.js').

Performance — Throttle & Debounce

Some events fire hundreds of times per second — scroll, resize, keypress. Running expensive code on every single event kills performance. Throttle and debounce are the two solutions.

The Problem — Event Flooding

The scroll event without any optimization
// BAD — scroll fires up to 100+ times/second while scrolling
// Each call does expensive DOM work → janky UI
window.addEventListener("scroll", () => {
  // This runs 100+ times per second while scrolling
  document.getElementById("pos").textContent = window.scrollY + "px";
  calculateVisibleElements();   // expensive!
  fetchMoreContent();           // network call on every pixel!
});

Debounce — Wait Until They Stop

Debounce delays execution until the user has stopped triggering the event for a given period. Perfect for search inputs — you don't want to fire an API call on every keystroke, only when the user pauses.

debounce.js — implement from scratch
function debounce(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);          // cancel any pending call
    timer = setTimeout(() => {
      fn.apply(this, args);       // call after delay
    }, delay);
  };
}

// Usage — search input
const searchInput = document.getElementById("search");

const handleSearch = debounce((event) => {
  console.log("Searching for:", event.target.value);
  // fetch(`/api/search?q=${event.target.value}`)
}, 400);  // wait 400ms after last keystroke

searchInput.addEventListener("input", handleSearch);

// Without debounce: 10 API calls typing "javascript"
// With debounce:    1 API call after typing stops

Throttle — Limit to Once Per Interval

Throttle ensures a function runs at most once per time interval, even if the event fires constantly. Perfect for scroll and resize handlers — you want them to run, but not on every single pixel.

throttle.js — implement from scratch
function throttle(fn, limit) {
  let inThrottle = false;
  return function(...args) {
    if (!inThrottle) {
      fn.apply(this, args);       // call immediately
      inThrottle = true;
      setTimeout(() => {
        inThrottle = false;       // allow next call after limit
      }, limit);
    }
  };
}

// Usage — scroll handler
const handleScroll = throttle(() => {
  console.log("Scroll position:", window.scrollY);
  updateStickyHeader();    // DOM work — safe to do every 100ms
}, 100);  // max once every 100ms

window.addEventListener("scroll", handleScroll);

// Without throttle: 100 calls/sec while scrolling
// With throttle:    10 calls/sec (100ms interval)

Which to Use?

Use Debounce when…

You only care about the final value after the user stops. Search inputs, form validation on blur, auto-save after the user stops typing, window resize layout recalculation.

Use Throttle when…

You want regular updates during a continuous action. Scroll position tracking, infinite scroll trigger, progress bars during upload, resize-based animations.

Lazy Loading — Load Only What's Needed

Don't load resources until they're needed. The simplest form is the native loading="lazy" attribute on images — the browser automatically defers loading images outside the viewport.

Lazy loading approaches
// 1. Native lazy loading — just add loading="lazy" to img
// <img src="photo.jpg" loading="lazy" alt="..." />

// 2. Intersection Observer — load when element enters viewport
const observer = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const img = entry.target;
      img.src = img.dataset.src;   // set real src
      observer.unobserve(img);     // stop watching
    }
  });
}, { threshold: 0.1 });  // trigger when 10% visible

document.querySelectorAll("img[data-src]").forEach(img => {
  observer.observe(img);
});

// 3. Avoid layout thrashing — read then write, not interleaved
// BAD — thrashes (forces reflow on each iteration)
elements.forEach(el => {
  const height = el.offsetHeight;       // READ — forces reflow
  el.style.height = (height * 2) + "px"; // WRITE
});

// GOOD — batch reads, then batch writes
const heights = elements.map(el => el.offsetHeight);  // all reads
elements.forEach((el, i) => {
  el.style.height = (heights[i] * 2) + "px";          // all writes
});

Try It — Debounce & Throttle

Try it
Memory aid: Debounce = "call me only after you stop." Throttle = "call me at most once per interval." Search box → debounce. Scroll handler → throttle.

Design Patterns — Module & Singleton

Design patterns are proven solutions to recurring code organisation problems. They don't have to be complex — at their core, they're just smarter ways to structure your code.

Why Design Patterns?

🏗️

As code grows, problems emerge: variables pollute the global scope, the same logic is duplicated everywhere, state becomes unpredictable. Design patterns give you a vocabulary and structure to solve these problems before they become debt.

Module Pattern — Private State with Public API

The Module pattern uses an IIFE (Immediately Invoked Function Expression) to create a private scope. Only what you explicitly return is public. Everything else is hidden.

module-pattern.js
// IIFE-based Module Pattern
const CartModule = (function() {
  // Private — not accessible outside
  const _items   = [];
  let   _discount = 0;

  function _calculateTotal() {
    const sub = _items.reduce((sum, item) => sum + item.price * item.qty, 0);
    return sub * (1 - _discount);
  }

  // Public API — only these are returned
  return {
    addItem(item) {
      _items.push(item);
      console.log(`Added: ${item.name}`);
    },
    removeItem(name) {
      const idx = _items.findIndex(i => i.name === name);
      if (idx !== -1) _items.splice(idx, 1);
    },
    setDiscount(pct) {
      _discount = pct / 100;
    },
    getTotal() {
      return _calculateTotal();
    },
    getItemCount() {
      return _items.length;
    }
  };
})();

CartModule.addItem({ name: "Laptop", price: 50000, qty: 1 });
CartModule.addItem({ name: "Mouse",  price: 800,   qty: 2 });
CartModule.setDiscount(10);  // 10% off

console.log("Items:", CartModule.getItemCount());  // 2
console.log("Total:", CartModule.getTotal());      // 46440

// Private state is inaccessible:
console.log(CartModule._items);    // undefined — good!
console.log(CartModule._discount); // undefined — good!

ES6 Module Equivalent

In real projects, use native ES6 modules (import/export) — the browser enforces module scope automatically. The IIFE pattern is still useful in single-file scripts and legacy environments.

cart.js — ES6 module (requires type="module" on the script tag)
// cart.js
const _items = [];   // private to this module

export function addItem(item) {
  _items.push(item);
}

export function getTotal() {
  return _items.reduce((sum, i) => sum + i.price * i.qty, 0);
}

Singleton Pattern — One Instance Only

A Singleton ensures only one instance of a class ever exists. Perfect for shared resources: a configuration manager, a connection pool, a global event bus.

singleton.js
class AppConfig {
  constructor() {
    if (AppConfig._instance) {
      // If instance exists, return it instead of creating new
      return AppConfig._instance;
    }
    // First-time initialisation
    this.theme    = "light";
    this.language = "en";
    this.apiUrl   = "https://api.example.com";
    AppConfig._instance = this;
  }

  set(key, value) {
    this[key] = value;
  }

  get(key) {
    return this[key];
  }
}

// Both variables point to the exact same object
const config1 = new AppConfig();
const config2 = new AppConfig();

config1.set("theme", "dark");

console.log(config2.get("theme"));       // "dark" — same instance!
console.log(config1 === config2);        // true — literally the same object

// Practical use — a Logger singleton
class Logger {
  constructor() {
    if (Logger._instance) return Logger._instance;
    this._logs = [];
    Logger._instance = this;
  }

  log(message) {
    const entry = { time: Date.now(), message };
    this._logs.push(entry);
    console.log(`[LOG] ${message}`);
  }

  getLogs() { return [...this._logs]; }
}

const logA = new Logger();
const logB = new Logger();

logA.log("App started");
logB.log("User logged in");

console.log(logA.getLogs().length);   // 2 — both went to the same logger
console.log(logA === logB);           // true

Design Patterns — Observer & Factory

Two more essential patterns: Observer lets parts of your app react to events without tight coupling, and Factory makes creating objects cleaner and more flexible.

Observer / PubSub Pattern

The Observer pattern lets objects (subscribers) listen for events emitted by another object (the publisher). Parts of your app communicate without directly knowing about each other.

event-emitter.js — Observer / PubSub from scratch
class EventEmitter {
  constructor() {
    this._listeners = {};  // { eventName: [fn, fn, ...] }
  }

  // Subscribe to an event
  on(event, listener) {
    if (!this._listeners[event]) {
      this._listeners[event] = [];
    }
    this._listeners[event].push(listener);
    return this;  // enable chaining
  }

  // Unsubscribe
  off(event, listener) {
    if (!this._listeners[event]) return;
    this._listeners[event] = this._listeners[event]
      .filter(fn => fn !== listener);
    return this;
  }

  // Subscribe once — auto-removes after first call
  once(event, listener) {
    const wrapper = (...args) => {
      listener(...args);
      this.off(event, wrapper);
    };
    return this.on(event, wrapper);
  }

  // Emit — call all listeners for an event
  emit(event, ...args) {
    if (this._listeners[event]) {
      this._listeners[event].forEach(fn => fn(...args));
    }
    return this;
  }
}

// Real-world example: a shopping cart event bus
const cart = new EventEmitter();

cart.on("item:added", (item) => {
  console.log(`Cart: Added ${item.name} (₹${item.price})`);
});

cart.on("item:added", (item) => {
  console.log(`Analytics: tracked add-to-cart for ${item.name}`);
});

cart.once("checkout", () => {
  console.log("Payment modal opened (won't open again)");
});

// Dispatch events
cart.emit("item:added", { name: "Laptop", price: 50000 });
cart.emit("item:added", { name: "Mouse",  price: 800   });
cart.emit("checkout");
cart.emit("checkout");  // once() listener won't fire again

// Output:
// Cart: Added Laptop (₹50000)
// Analytics: tracked add-to-cart for Laptop
// Cart: Added Mouse (₹800)
// Analytics: tracked add-to-cart for Mouse
// Payment modal opened (won't open again)

Factory Pattern

A Factory is a function or class that creates objects for you. Instead of new SomeClass() scattered everywhere, you call a factory — it decides what type to create based on the input.

factory-pattern.js
// Without factory — callers need to know all class names
// const shape = new Circle(5);
// const rect  = new Rectangle(4, 6);

// With factory — one entry point, internal logic hidden
class Circle {
  constructor(radius) {
    this.type   = "circle";
    this.radius = radius;
  }
  area()      { return Math.PI * this.radius ** 2; }
  perimeter() { return 2 * Math.PI * this.radius; }
}

class Rectangle {
  constructor(width, height) {
    this.type   = "rectangle";
    this.width  = width;
    this.height = height;
  }
  area()      { return this.width * this.height; }
  perimeter() { return 2 * (this.width + this.height); }
}

class Triangle {
  constructor(a, b, c) {
    this.type = "triangle";
    this.a = a; this.b = b; this.c = c;
  }
  perimeter() { return this.a + this.b + this.c; }
  area() {
    const s = this.perimeter() / 2;
    return Math.sqrt(s * (s-this.a) * (s-this.b) * (s-this.c));
  }
}

// The Factory
function ShapeFactory(type, ...args) {
  switch (type) {
    case "circle":    return new Circle(...args);
    case "rectangle": return new Rectangle(...args);
    case "triangle":  return new Triangle(...args);
    default:          throw new Error(`Unknown shape: ${type}`);
  }
}

// Callers only use ShapeFactory — no class names needed
const c = ShapeFactory("circle", 7);
const r = ShapeFactory("rectangle", 4, 6);
const t = ShapeFactory("triangle", 3, 4, 5);

console.log(`Circle   area: ${c.area().toFixed(2)}`);     // 153.94
console.log(`Rectangle area: ${r.area()}`);               // 24
console.log(`Triangle  area: ${t.area().toFixed(2)}`);    // 6.00
console.log(`Triangle  perim: ${t.perimeter()}`);         // 12

// Adding a new shape later only changes the factory — not every caller

Factory in the wild

class ValidationError extends Error {
  constructor(message, field) {
    super(message);
    this.name  = "ValidationError";
    this.field = field;
  }
}

const PATTERNS = {
  name:  /^[a-zA-Z ]{2,}$/,
  email: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
  phone: /^[6-9]\d{9}$/,
  pin:   /^\d{6}$/
};

function validateField(key, value, pattern, errorMsg) {
  if (!pattern.test((value || "").trim())) {
    throw new ValidationError(errorMsg, key);
  }
  return value.trim();
}

function validateForm(data) {
  const results = {};
  const errors  = [];

  const fields = [
    { key: "name",  pattern: PATTERNS.name,  msg: "Name must be 2+ letters only" },
    { key: "email", pattern: PATTERNS.email, msg: "Enter a valid email address" },
    { key: "phone", pattern: PATTERNS.phone, msg: "Phone must be 10 digits starting with 6-9" },
    { key: "pin",   pattern: PATTERNS.pin,   msg: "PIN must be exactly 6 digits" },
  ];

  for (const f of fields) {
    try {
      results[f.key] = {
        value:  validateField(f.key, data[f.key] || "", f.pattern, f.msg),
        status: "VALID"
      };
    } catch (e) {
      if (e instanceof ValidationError) {
        results[f.key] = { value: data[f.key], status: "INVALID", error: e.message };
        errors.push(f.key);
      } else {
        throw e;
      }
    }
  }

  return { results, isValid: errors.length === 0 };
}

function saveFormData(data) {
  try {
    localStorage.setItem("lastForm", JSON.stringify(data));
    console.log("Saved to localStorage");
  } catch (e) {
    console.error("Save failed:", e.message);
  }
}

  try {
    const raw = localStorage.getItem("lastForm");
    return raw ? JSON.parse(raw) : null;
  } catch (e) {
    console.error("Load failed:", e.message);
    return null;
  }
}
